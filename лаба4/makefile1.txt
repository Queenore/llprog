.PHONY: all clean

OBJS= gorn.c

AR = riscv64-unknown-elf-ar.exe
CC = riscv64-unknown-elf-gcc.exe

MYLIBNAME = lib.a

CFLAGS= -O1

INCLUDES+= -I .

vpath %.h .
vpath %.c .

%.o: %.c
	$(CC) -MD $(CFLAGS) $(INCLUDES) -c $< -o $@

all: $(MYLIBNAME)

$(MYLIBNAME): gorn.o
	$(AR) -rsc $@ $^
